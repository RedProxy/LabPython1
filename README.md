        ```bash
        print("Program Menentukan Bilangan Terbesar")
        print("====================================")

        a = int(input("Masukan nilai a: "))
        b = int(input("Masukan nilai b: "))

        if a > b:
            print ("A yang terbesar")
        elif b > a:
            print ("B yang terbesar")
        ```
